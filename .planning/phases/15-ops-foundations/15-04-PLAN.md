Create web/lib/server/logger.ts. This is a thin synchronous wrapper around console.error that adds structured fields. No external dependencies, no async, no network calls.

```typescript
/**
 * Structured server-side error logger for API routes.
 * Output goes to stdout (captured by Vercel function logs).
 * Fields are JSON-serializable for log filtering.
 */

export interface ApiErrorContext {
  /** Full route path, e.g. "/api/charting/games/123" */
  route: string;
  /** HTTP method, e.g. "GET", "POST", "PATCH", "DELETE" */
  method: string;
  /** Brief description of what was being attempted, e.g. "fetch game" */
  action: string;
  /** The caught error */
  error: unknown;
  /** Optional: request-scoped context (e.g. gameId, userId) */
  context?: Record<string, unknown>;
}

export function logApiError({
  route,
  method,
  action,
  error,
  context,
}: ApiErrorContext): void {
  const message = error instanceof Error ? error.message : String(error);
  const stack = error instanceof Error ? error.stack : undefined;

  console.error(
    JSON.stringify({
      level: "error",
      route,
      method,
      action,
      message,
      stack,
      context: context ?? null,
      timestamp: new Date().toISOString(),
    }),
  );
}
```
