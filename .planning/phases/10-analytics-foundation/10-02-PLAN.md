const BREAKING_TYPES: PitchType[] = ["Curveball", "Slider"];

function computePitchGroup(pitches: ChartingPitch[], types: PitchType[]): PitchGroupStats {
  const group = pitches.filter(p => types.includes(p.pitchType));
  const swings = group.filter(p => SWING_RESULTS.includes(p.pitchResult)).length;
  const whiffs = group.filter(p => p.pitchResult === "swinging_strike").length;
  return { pitches: group.length, swings, whiffs, whiffPct: pct(whiffs, swings) };
}

**computeHitterStats(hitterName, gameId):**
1. Fetch all PAs where `gameId = gameId` from `chartingPlateAppearances`
2. Filter in JS to `pa.hitterName === hitterName`
3. If none: return null
4. Fetch pitches where `paId IN paIds`
5. Call `computeHitterStats_pure(pitches, hitterPas)`

**aggregateHitterStats(hitterName, options?):**
1. If `options?.gameIds` provided: query PAs where `gameId IN gameIds`
2. If `options?.from` or `options?.to` provided: first resolve matching game IDs from `chartingGames` filtered by `gameDate` (ISO string comparison), then query PAs where `gameId IN filteredGameIds`
3. Otherwise: query ALL `chartingPlateAppearances` rows (no game filter) — then filter in JS by `hitterName`
4. Count unique `segmentId` values across matching PAs as the session proxy (or unique gameIds — use gameId as the "session" boundary for hitters since they span multiple segments)
5. Fetch pitches for all collected PA ids
6. Call `computeHitterAggregation(allPitches, hitterPas, uniqueGameCount)`

Note: Fetching all plate appearances without a game filter is acceptable for this dataset size. Add a comment noting this should be revisited if the dataset grows beyond ~10k rows.
