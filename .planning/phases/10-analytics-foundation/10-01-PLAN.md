**computeSegmentStats(segmentId: string):**
```typescript
export async function computeSegmentStats(segmentId: string): Promise<SegmentStats | null> {
  // 1. Fetch PAs for this segment
  const pas = await db.select().from(chartingPlateAppearances)
    .where(eq(chartingPlateAppearances.segmentId, segmentId));
  if (pas.length === 0) return null;
  // 2. Fetch pitches for those PAs
  const paIds = pas.map(pa => pa.id);
  const pitches = await db.select().from(chartingPitches)
    .where(inArray(chartingPitches.paId, paIds));
  return computeSegmentStats_pure(pitches, pas);
}
```

**computePitcherAggregation (pure, exported for tests):**
```typescript
export function computePitcherAggregation(
  allPitches: ChartingPitch[],
  allPas: ChartingPlateAppearance[],
  sessions: number
): AggregatedPitcherStats | null {
  const base = computeSegmentStats_pure(allPitches, allPas);
  if (!base) return null;
  return { ...base, sessions, totalPAs: allPas.length };
}
```
