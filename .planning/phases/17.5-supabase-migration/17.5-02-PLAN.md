<!-- withTeam target interface -->
```typescript
// web/db/withTeam.ts
import { db } from "./index";
import postgres from "postgres";

// teamId: e.g. 'babson' — lowercase slug, not display name
export async function withTeam<T>(
  teamId: string,
  callback: (db: typeof import("./index").db) => Promise<T>
): Promise<T> {
  // Use a dedicated connection for the transaction so SET LOCAL is scoped
  return db.transaction(async (tx) => {
    await tx.execute(sql`SET LOCAL app.team_id = ${teamId}`);
    return callback(tx as unknown as typeof db);
  });
}
```

Note: postgres-js (Plan 01) supports db.transaction() — use it here. SET LOCAL scopes the variable to the transaction, which is exactly what RLS requires.

    **Step 1: Create web/db/withTeam.ts**

    ```typescript
    import { sql } from "drizzle-orm";
    import { db } from "./index";

    /**
     * Wraps a database callback in a transaction that sets the RLS team_id
     * session variable before executing any queries.
     *
     * All charting table access MUST go through withTeam — RLS will return
     * 0 rows for any query that doesn't set app.team_id first.
     *
     * @param teamId - Lowercase team slug, e.g. 'babson'
     * @param callback - Receives the transaction handle to use for queries
     */
    export async function withTeam<T>(
      teamId: string,
      callback: (tx: Parameters<Parameters<typeof db.transaction>[0]>[0]) => Promise<T>
    ): Promise<T> {
      return db.transaction(async (tx) => {
        await tx.execute(sql`SET LOCAL app.team_id = ${teamId}`);
        return callback(tx);
      });
    }
    ```

    **Step 3: Update all five charting API routes**

    For each route file, wrap ALL database operations in `withTeam(TEAM_ID, async (tx) => { ... })`.

    Pattern for GET routes (example — games list):
    ```typescript
    import { withTeam } from "@/db/withTeam";
    import { TEAM_ID } from "@/lib/teamConfig";

    // Before:
    const games = await db.select().from(chartingGames).orderBy(...);

    // After:
    const games = await withTeam(TEAM_ID, async (tx) => {
      return tx.select().from(chartingGames).orderBy(...);
    });
    ```

    Pattern for routes using db.batch() (e.g., game PATCH in [id]/route.ts):
    ```typescript
    // Before:
    await db.batch([
      db.update(chartingGames).set({...}).where(eq(chartingGames.id, id)),
      db.insert(chartingPitches).values({...}),
    ]);

    // After:
    await withTeam(TEAM_ID, async (tx) => {
      // db.batch() is not available on transaction handles — use sequential awaits
      await tx.update(chartingGames).set({...}).where(eq(chartingGames.id, id));
      await tx.insert(chartingPitches).values({...});
    });
    ```

    Note: db.batch() on the postgres-js adapter is not the same as neon's batch API. Inside a transaction, use sequential awaits — the transaction itself provides atomicity.
