    ```bash
    #!/usr/bin/env bash
    # Publish a completed outing's assets to Supabase Storage.
    # Usage: ./scripts/publish_to_storage.sh <playerId> <dateId>
    # Example: ./scripts/publish_to_storage.sh JFinkelstein1 2025_04_27
    #
    # Prerequisites:
    #   - supabase CLI installed and linked (supabase link --project-ref ojrkhiatugtrwavyeakk)
    #   - Outing files processed and in outings/<playerId>/<dateId>/

    set -euo pipefail

    PLAYER_ID="${1:?Usage: $0 <playerId> <dateId>}"
    DATE_ID="${2:?Usage: $0 <playerId> <dateId>}"
    BUCKET="outings"
    LOCAL_DIR="outings/${PLAYER_ID}/${DATE_ID}"
    REMOTE_PATH="ss://${BUCKET}/${PLAYER_ID}/${DATE_ID}"

    echo "Publishing ${PLAYER_ID}/${DATE_ID} to Supabase Storage..."

    # Verify source files exist
    if [ ! -d "${LOCAL_DIR}" ]; then
      echo "ERROR: Source directory ${LOCAL_DIR} does not exist"
      exit 1
    fi

    if [ ! -f "${LOCAL_DIR}/pitch_data_overlay_lite.csv" ]; then
      echo "ERROR: Missing pitch_data_overlay_lite.csv in ${LOCAL_DIR}"
      exit 1
    fi

    # Count validation
    CSV_ROWS=$(tail -n +2 "${LOCAL_DIR}/pitch_data_overlay_lite.csv" | wc -l | xargs)
    CLIP_COUNT=$(ls "${LOCAL_DIR}/clips/pitch_"*.mp4 2>/dev/null | wc -l | xargs)
    OVERLAY_COUNT=$(ls "${LOCAL_DIR}/results/pitch_"*_overlay.mp4 2>/dev/null | wc -l | xargs)

    echo "CSV rows: ${CSV_ROWS} | Clips: ${CLIP_COUNT} | Overlays: ${OVERLAY_COUNT}"

    if [ "${CSV_ROWS}" != "${CLIP_COUNT}" ] || [ "${CSV_ROWS}" != "${OVERLAY_COUNT}" ]; then
      echo "ERROR: File counts do not match. Aborting."
      exit 1
    fi

    # Upload CSV
    echo "Uploading CSV..."
    supabase storage cp "${LOCAL_DIR}/pitch_data_overlay_lite.csv" \
      "${REMOTE_PATH}/pitch_data_overlay_lite.csv"

    # Upload clips
    echo "Uploading clips (${CLIP_COUNT} files)..."
    supabase storage cp -r "${LOCAL_DIR}/clips/" "${REMOTE_PATH}/clips/"

    # Upload overlays
    echo "Uploading overlays (${OVERLAY_COUNT} files)..."
    supabase storage cp -r "${LOCAL_DIR}/results/" "${REMOTE_PATH}/results/"

    echo ""
    echo "Done. Files are at:"
    echo "  https://ojrkhiatugtrwavyeakk.supabase.co/storage/v1/object/public/${BUCKET}/${PLAYER_ID}/${DATE_ID}/"
    echo ""
    echo "Next steps:"
    echo "  1. Update web/lib/dataIndex.ts — add outing entry using buildDataPaths('${PLAYER_ID}', '${DATE_ID}')"
    echo "  2. npm --prefix web run build"
    echo "  3. git add web/lib/dataIndex.ts && git commit && git push"
    ```

    Make the script executable:
    ```bash
    chmod +x scripts/publish_to_storage.sh
    ```

    Note: The script does NOT update dataIndex.ts automatically — that step remains manual (Plan 04 makes buildDataPaths return Supabase URLs so dataIndex.ts entries are trivially correct once Plan 04 ships).

    Note: Video files in `web/public/data/` that are already committed to git can remain there for now. The existing published outings will continue to work via static serving until dataIndex.ts is updated to point to Supabase URLs (Plan 04). There is no forced cutover — existing entries can be migrated incrementally.
